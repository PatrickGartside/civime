POST /api/v1/reminders:
  summary: Create a meeting reminder
  auth: X-API-Key required
  body: { email: string, meeting_id: integer }
  response 201: { id, email, meeting_id, status: "pending_confirmation" }
  response 409: { error: "Reminder already exists" }
  side_effect: Sends confirmation email with confirm_token link

GET /api/v1/reminders/confirm:
  summary: Confirm a reminder opt-in
  auth: Public (token in query param)
  query: { token: string }  (64-char hex confirm_token from reminders table)
  response: 302 redirect to https://civi.me/meetings/{id}/notify/?confirmed=1
  side_effect: Sets reminders.confirmed=1
